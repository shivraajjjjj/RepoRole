2️⃣ Generic Scoring Function Pattern

This is the key abstraction.

function scoreCategory(categoryConfig, detectedSignals):
    score = 0

    for each signal in categoryConfig:
        if detectedSignals contains signal:
            score += categoryConfig[signal]

    return min(score, categoryConfig.cap)

🟦 Languages
scoreLanguages():
    return scoreCategory(
        role.weights.languages,
        repo.languages
    )



🟦 Runtime (if exists)
scoreRuntime():
    if role.weights.runtime exists:
        return scoreCategory(
            role.weights.runtime,
            repo.runtime
        )
    return 0


🟦 Frameworks (core + optional)
scoreFrameworks():
    score = 0

    if role.weights.frameworks.core exists:
        score += scoreCategory(
            role.weights.frameworks.core,
            repo.frameworks
        )

    if role.weights.frameworks.optional exists:
        score += scoreCategory(
            role.weights.frameworks.optional,
            repo.frameworks
        )

    return min(score, role.weights.frameworks.cap)


🟦 Structure
scoreStructure():
    return scoreCategory(
        role.weights.structure,
        repo.folders
    )


🟦 Database / Persistence
scoreDatabase():
    if repo.databases.count >= 1:
        score = role.weights.database.any

    if repo.databases.count > 1:
        score = role.weights.database.multiple

    return min(score, role.weights.database.cap)

🟦 Build / Dependencies (role-specific)
scoreBuildOrDeps():
    if role.weights.build exists:
        return scoreCategory(
            role.weights.build,
            repo.buildFiles
        )

    if role.weights.dependencies exists:
        return scoreCategory(
            role.weights.dependencies,
            repo.dependencyFiles
        )

    return 0

🟦 Infra / Quality
scoreInfra():
    return scoreCategory(
        role.weights.infra,
        repo.infraSignals
    )


4️⃣ Penalty Engine (IMPORTANT)
applyPenalties():
    penalty = 0

    if repo.isTutorial:
        penalty += tutorialPenaltyScore   // computed separately

    return min(penalty, role.weights.penalties.tutorial_max)
